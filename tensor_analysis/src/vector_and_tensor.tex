\chapter{矢量与张量}

\section{矢量的基本运算}

\subsection{叉积}

\begin{align}
\ten{w} &= \ten{u} \times \ten{v}\cr
~ &= \begin{vmatrix}
\ten{i} & \ten{j} & \ten{k}\cr
u_x & u_y & u_z\cr
v_x & v_y & v_z\end{vmatrix}
\end{align}

\subsection{混合积}

\begin{align}
\left[\ten{u\ v\ w}\right] &= (\ten{u}\times\ten{v})\cdot\ten{w} = \ten{u}\cdot(\ten{v}\times\ten{w})\cr
~&= \begin{vmatrix}
u_x & u_y & u_z\cr
v_x & v_y & v_z\cr
w_x & w_y & w_z\end{vmatrix}
\end{align}
