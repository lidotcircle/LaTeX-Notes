\chapter{土的性质}

\section{Physical parameters of soil}

%{{{ 基本指标
孔隙率(\engterm{void ratio}):
\begin{equation}
\index{孔隙率:e}
e={V_v\over V_s}\newline
\end{equation}

饱和度(\engterm{Degree of saturation}):
\begin{equation}
\index{饱和度:S}
S={V_w\over V_v}
\end{equation}

含水率(\engterm{Moisture content}):
\begin{equation}
\index{含水率:$\omega$}
\omega={W_s\over W_w}
\end{equation}

重度(\engterm{unit weight of soil}):
\begin{equation}
\index{重度:$\gamma$}
\gamma={W\over V}
\end{equation}

干重度(\engterm{dry unit weight of soil}):
\begin{equation}
\index{干重度:$\omega_d$}
\gamma_d={W_s\over V}
\end{equation}

饱和重度(\engterm{saturated unit weight of soil}):
\begin{equation}
\index{饱和重度:$\gamma_{sat}$}
\gamma_{sat}={{\gamma_wG_s\left(1+\omega\right)}\over{1+e}}=%
{{\gamma_w\left(G_s+\omega e/\omega\right)}\over{1+e}}%
={{\gamma_w\left(G_s+e\right)}\over{1+e}}
\end{equation}

比重(\engterm{spcific gravity}):
\begin{equation}
\index{比重:$G_s$}
G_s = {\gamma_s\over\gamma_w}
\end{equation}

相对密度(\engterm{relative density}):
\begin{equation}
\index{相对密度:$D_r$}
D_r = {e_{max} - e\over{e_{max} - e_{min}}}
\end{equation}
%}}} End : 基本指标
